#include <stdio.h>

/* 参数array_l 和array_r 表示的是int 数组的左右区间的指针，区间
 * 左闭右开，参数val 是要找的值。返回值是大于等于val 的第一个
 * 数组元素对应的指针。*/
int *lower_bound(int *array_l, int *array_r, int val) {
    if(array_l >= array_r) {
        fprintf(stderr, "lower_bound error: array_l greater equal than array_r\n");
        return NULL;
    }
    while(array_l < array_r) {
        int *mid = array_l + (array_r - array_l) / 2;
        if(*mid >= val) {
            array_r = mid;
        } else {
            array_l = mid + 1;
        }
    }
    return array_r;
}
